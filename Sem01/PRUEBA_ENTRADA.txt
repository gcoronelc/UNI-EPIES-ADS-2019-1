SELECT 
  COUNT(1) CANT_VENTAS,
  SUM(IMPORTE) IMPORTE_TOTAL
FROM VENTA 
GROUP BY CIUDAD;


2.-

SELECT 
   P.CATEGORIA,
   SUM(V.CANTIDAD) CANTIDAD,
   SUM(V.IMPORTE)  IMPORTE
FROM CATEGORIA C 
JOIN VENTA V
ON C.IDPRODUCTO = V.IDPRODUCTO
GROUP BY P.CATEGORIA;

3.-

SELECT
   V.CIUDAD, 
   P.CATEGORIA,
   SUM(V.CANTIDAD) CANTIDAD,
   SUM(V.IMPORTE)  IMPORTE
FROM CATEGORIA C 
JOIN VENTA V
ON C.IDPRODUCTO = V.IDPRODUCTO
GROUP BY V.CIUDAD, P.CATEGORIA;
